1   package pacman.objects;
2   
3   import pacman.framework.GameObject;
4   import java.awt.Color;
5   import java.awt.Graphics;
6   import pacman.framework.Direction;
7   
8   public class Player extends GameObject {
9       
10      public Player(int x, int y) {
11          super(x, y);
12          width = 24;
13          height = 24;
14          
15          velocity = 3;
16          
17          this.direction = new Direction("left");
18      }
19      
20      @Override
21      public void update() {
22          if (direction.equals("left")) {
23              this.x -= velocity;
24          } else if (direction.equals("down")) {
25              this.y += velocity;
26          } else if (direction.equals("right")) {
27              this.x += velocity;
28          } else if (direction.equals("up")) {
29              this.y -= velocity;
30          }
31      }
32      
33      @Override
34      public void render(Graphics g) {
35          g.setColor(Color.yellow);
36          g.fillRect(x, y, width, height);
37      }
38  }
