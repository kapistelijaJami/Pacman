1   package pacman.game;
2   
3   import java.awt.Graphics;
4   import java.awt.image.BufferedImage;
5   import pacman.framework.Tile;
6   import pacman.imagehandling.ImageLoader;
7   import pacman.objects.Player;
8   
9   public class Level {
10      private Tile[][] tiles;
11      private BufferedImage levelImage;
12      
13      public Level(int width, int height) { //width ja height Tile koossa (ei pikseleissä)
14          tiles = new Tile[height][width];
15      }
16      
17      public void loadLevelImage(String path) {
18          ImageLoader loader = new ImageLoader();
19          levelImage = loader.loadImage(path);
20      }
21      
22      public BufferedImage getLevelImage() {
23          return this.levelImage;
24      }
25      
26      public Tile[][] getTiles() {
27          return this.tiles;
28      }
29      
30      public void makeLevelFromImage(Pacman game) {
31          int w = levelImage.getWidth();
32          int h = levelImage.getHeight();
33  
34          
35          for (int yy = 0; yy < h; yy++) {
36              for (int xx = 0; xx < w; xx++) { //käydään läpi rivi kerrallaan
37                  int pixel = levelImage.getRGB(xx, yy);
38                  int red = (pixel >> 16) & 0xff;
39                  int green = (pixel >> 8) & 0xff;
40                  int blue = (pixel) & 0xff;
41  
42                  //KAIKKI SEINÄT
43                  if (red == 0 && green == 0 && blue == 255) { //pystysuora
44                      tiles[yy][xx] = new Tile(xx * Pacman.TILE_WIDTH, yy * Pacman.TILE_HEIGHT);
45                      tiles[yy][xx].setIsWall(true);
46                      
47                  } else if (red == 0 && green == 0 && blue == 200) { //vaakasuora
48                      tiles[yy][xx] = new Tile(xx * Pacman.TILE_WIDTH, yy * Pacman.TILE_HEIGHT);
49                      tiles[yy][xx].setIsWall(true);
50                      
51                  } else if (red == 0 && green == 0 && blue == 120) { //kulmakohta
52                      //TÄSSÄ PÄÄTTELE MITEN PÄIN KULMA TULEE
53                      tiles[yy][xx] = new Tile(xx * Pacman.TILE_WIDTH, yy * Pacman.TILE_HEIGHT);
54                      tiles[yy][xx].setIsWall(true);
55                      
56                  } else if (red == 0 && green == 0 && blue == 70) { //reuna
57                      tiles[yy][xx] = new Tile(xx * Pacman.TILE_WIDTH, yy * Pacman.TILE_HEIGHT);
58                      tiles[yy][xx].setIsWall(true);
59                      
60                  } else if (red == 0 && green == 0 && blue == 50) { //reunan kulmat
61                      tiles[yy][xx] = new Tile(xx * Pacman.TILE_WIDTH, yy * Pacman.TILE_HEIGHT);
62                      tiles[yy][xx].setIsWall(true);
63                  }
64                  
65                  
66                  //SYÖTÄVÄT
67                  if (red == 130 && green == 130 && blue == 130) { //pieni pallo
68                      tiles[yy][xx] = new Tile(xx * Pacman.TILE_WIDTH, yy * Pacman.TILE_HEIGHT);
69                      tiles[yy][xx].setIsFood(true);
70                      
71                  } else if (red == 255 && green == 255 && blue == 255) { //iso pallo
72                      tiles[yy][xx] = new Tile(xx * Pacman.TILE_WIDTH, yy * Pacman.TILE_HEIGHT);
73                      tiles[yy][xx].setIsEnergizer(true);
74                  }
75                  
76  
77                  //PLAYER
78                  if (red == 255 && green == 255 && blue == 0) {
79                      game.setPlayer(new Player(xx * Pacman.TILE_WIDTH + Pacman.TILE_WIDTH / 2, yy * Pacman.TILE_HEIGHT)); //luodaan pelaaja
80                  }
81                  
82                  
83                  //JOS EI OLLUT MIKÄÄN ERIKOINEN TILE, TAI LUOTIIN JOTAIN MUUTA, KUTEN PLAYER, NIIN TEHDÄÄN TYHJÄ TILE TÄSSÄ
84                  if (tiles[yy][xx] == null) {
85                      tiles[yy][xx] = new Tile(xx * Pacman.TILE_WIDTH, yy * Pacman.TILE_HEIGHT); //tehdään tyhjä tile
86                  }
87              }
88          }
89      }
90      
91      public void render(Graphics g) {
92          for (Tile[] row : tiles) {
93              for (Tile tile : row) {
94                  if (tile != null) {
95                      tile.render(g);
96                  }
97              }
98          }
99      }
100 }
