1   package pacman.game;
2   
3   import java.awt.Canvas;
4   import java.awt.Color;
5   import java.awt.Graphics;
6   import java.awt.image.BufferStrategy;
7   import pacman.framework.KeyInput;
8   import pacman.objects.Player;
9   
10  public class Pacman extends Canvas implements Runnable {
11      
12      public static final int WIDTH = 672;
13      public static final int HEIGHT = 744;
14      
15      private Thread thread;
16      private boolean running = false;
17      
18      public static Player player;
19      
20      public Pacman() {
21          new Window(WIDTH, HEIGHT, "Pacman", this);
22      }
23      
24      public void init() {
25          player = new Player(WIDTH / 2, HEIGHT / 2); //luodaan pelaaja
26          
27          this.addKeyListener(new KeyInput(player)); //luodaan keyListener
28      }
29      
30      public synchronized void start() {
31          if (running) {
32              return;
33          }
34          
35          init(); //tehdään kartta ja luodaan peliobjektit ja keyListener
36          this.requestFocus(); //ei tarvitse erikseen klikata ikkunaa jotta kontrollit toimisivat
37          
38          running = true;
39          thread = new Thread(this);
40          thread.start();
41      }
42      
43      public synchronized void stop() {
44          try {
45              thread.join(); //pysäyttää threadin
46              running = false;
47          } catch (Exception e) {
48          }
49      }
50  
51      @Override
52      public void run() {
53          long timer = System.currentTimeMillis();
54          long lastTime = System.nanoTime();
55          double updatesPerSecond = 60.0;
56          double nanoSecondsPerUpdate = 1000000000 / updatesPerSecond; //päivitysten väliin tarvittavat nanosekunnit
57          double delta = 0; //nanosekuntien määrä suhteessa päivitykseen tarvittaviin nanosekunteihin, kun 1, niin tehdään päivitys
58          
59          int updates = 0; //päivitysten määrä sekunnissa
60          int frames = 0; //renderöintien määrä sekunnissa
61          
62          while (running) { //pelisilmukka
63              long now = System.nanoTime();
64              delta += (now - lastTime) / nanoSecondsPerUpdate; //deltaan lisätään loopiin kuluneet nanosekunnit per päivitykseen tarvittavat nanosekunnit
65              lastTime = now;
66              
67              while (delta >= 1) { //deltan arvo on 1 kun on mennyt 1/updatesPerSecond sekuntia (eli 1/60)
68                  update(); //päivitetään peliä
69                  updates++;
70                  delta--; //deltan arvoa miinustetaan yhdellä (eli putoaa hyvin lähelle lukua 0)
71              }
72              render();
73              frames++;
74              
75              if (System.currentTimeMillis() - timer > 1000) { //jos on mennyt sekunti viime kerrasta (tänne pääsee siis sekunnin välein)
76                  timer += 1000;
77                  System.out.println("Updates: " + updates + ", Frames: " + frames); //näitä voi käyttää esim titlessä
78                  frames = 0;
79                  updates = 0;
80              }
81          }
82          stop();
83      }
84      
85      private void update() {
86          player.update();
87      }
88      
89      private void render() {
90          BufferStrategy bs = getBufferStrategy();
91          if (bs == null) {
92              createBufferStrategy(3);
93              return;
94          }
95          
96          Graphics g = bs.getDrawGraphics();
97          //piirrä tässä//
98          g.setColor(Color.black);
99          g.fillRect(0, 0, WIDTH, HEIGHT);
100         
101         player.render(g);
102         //piirto loppuu tähän//
103         g.dispose();
104         bs.show();
105     }
106 }
