241     /**
242      * Metodi alustaa pelisilmukan muuttujat ja käynnistää silmukan,
243      * jossa kutsutaan update ja render -metodeita.
244      */
245     @Override
246     public void run() {
247         long timer = System.currentTimeMillis();
248         long lastTime = System.nanoTime();
249         double nanoSecondsPerUpdate = 1000000000 / updatesPerSecond; //päivitysten väliin tarvittavat nanosekunnit
250         double delta = 0; //nanosekuntien määrä suhteessa päivitykseen tarvittaviin nanosekunteihin, kun 1, niin tehdään päivitys
251         
252         int updates = 0; //päivitysten määrä sekunnissa
253         int frames = 0; //renderöintien määrä sekunnissa
254         
255         while (running) { //pelisilmukka
256             /*update();
257             render();*/
258             long now = System.nanoTime();
259             delta += (now - lastTime) / nanoSecondsPerUpdate; //deltaan lisätään loopiin kuluneet nanosekunnit per päivitykseen tarvittavat nanosekunnit
260             lastTime = now;
261             
262             while (delta >= 1) { //deltan arvo on 1 kun on mennyt 1/updatesPerSecond sekuntia (eli 1/60)
263                 update(); //päivitetään peliä
264                 updates++;
265                 render();
266                 frames++;
267                 delta--; //deltan arvoa miinustetaan yhdellä (eli putoaa hyvin lähelle lukua 0)
268             }
269             
270             if (System.currentTimeMillis() - timer > 1000) { //jos on mennyt sekunti viime kerrasta (tänne pääsee siis sekunnin välein)
271                 timer += 1000; //timer jahtaa currentTimeMillis funktiota
272                 System.out.println("Updates: " + updates + ", Frames: " + frames); //näitä voi käyttää esim titlessä
273                 this.frame.setTitle("Pacman " + "Updates: " + updates + ", Frames: " + frames);
274                 frames = 0;
275                 updates = 0;
276             }
277         }
278         stop();
279     }

281     /**
282      * Metodi päivittää pelin tilannetta jokaiselle peliobjektille.
283      */
284     private void update() {
285         if (!paused) {
286             checkScatter();
287             ghostHandler.checkGetOut(level, gameUpdates);
288             
289             ghostHandler.update(level);
290             player.update(level);
291             
292             gameUpdates++;
293         }
294     }

296     /**
297      * Metodi piirtää kentän ja peliobjektit.
298      */
299     private void render() {
300         BufferStrategy bs = getBufferStrategy();
301         if (bs == null) {
302             createBufferStrategy(3);
303             return;
304         }
305         
306         Graphics g = bs.getDrawGraphics();
307         //piirrä tässä//
308         g.setColor(Color.black);
309         g.fillRect(0, 0, WIDTH, HEIGHT + 40); //musta tausta
310         
311         level.render(g);
312         ghostHandler.render(g);
313         player.render(g);
314         
315         g.setColor(Color.yellow);
316         g.setFont(new Font("Arial", Font.BOLD, 20));
317         g.drawString("Pisteet: " + player.getPoints(), 192, 764);
318         
319         for (int i = 0; i < player.getLives() - 1; i++) {
320             g.drawImage(lives, 24 + (48 * i), 740, null);
321         }
322         
323         if (gameOver) {
324             g.setColor(Color.red);
325             g.setFont(new Font("Arial", Font.BOLD, 22));
326             g.drawString("Game Over!", 275, 425);
327         }
328         
329         //piirto loppuu tähän//
330         g.dispose();
331         bs.show();
332     }
