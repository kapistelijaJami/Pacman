1   package pacman.ui;
2   
3   import java.awt.Dimension;
4   import javax.swing.JFrame;
5   import pacman.game.Pacman;
6   
7   /**
8    * Luokka luo ikkunan, johon peli piirretään.
9    */
10  public class Window {
11      private JFrame frame;
12      
13      public Window(int width, int height, String title, Pacman game) {
14          frame = new JFrame(title);
15          
16          frame.setPreferredSize(new Dimension(width, height)); //laitetaan ikkunan koko
17          frame.setMinimumSize(new Dimension(width, height));
18          frame.setMaximumSize(new Dimension(width, height));
19          
20          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); //ikkuna sulkeutuu rastista
21          frame.setResizable(false); //ikkunaa ei voida venyttää
22          frame.setLocationRelativeTo(null); //ikkuna syntyy näytön keskelle
23          frame.add(game); //lisätään peli ikkunaan
24          frame.pack(); //pakataan kaikki
25          frame.setVisible(true); //laitetaan ikkuna näkyväksi
26      }
27      
28      public JFrame getFrame() {
29          return this.frame;
30      }
31  }
