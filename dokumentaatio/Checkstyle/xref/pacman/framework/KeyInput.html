1   package pacman.framework;
2   
3   import java.awt.event.KeyAdapter;
4   import java.awt.event.KeyEvent;
5   import pacman.game.Pacman;
6   import pacman.objects.Player;
7   
8   /**
9    * Luokka huolehtii käyttäjän näppäinpainalluksista, joilla pelaajaa ohjataan.
10   */
11  public class KeyInput extends KeyAdapter {
12      private Player player;
13      
14      public KeyInput(Player player) {
15          this.player = player;
16      }
17      
18      @Override
19      public void keyPressed(KeyEvent e) {
20          int key = e.getKeyCode();
21          
22          if (key == KeyEvent.VK_LEFT || key == KeyEvent.VK_A) {
23              player.setNextDirection(Direction.LEFT);
24          } else if (key == KeyEvent.VK_DOWN || key == KeyEvent.VK_S) {
25              player.setNextDirection(Direction.DOWN);
26          } else if (key == KeyEvent.VK_RIGHT || key == KeyEvent.VK_D) {
27              player.setNextDirection(Direction.RIGHT);
28          } else if (key == KeyEvent.VK_UP || key == KeyEvent.VK_W) {
29              player.setNextDirection(Direction.UP);
30              
31          } else if (key == KeyEvent.VK_ENTER || key == KeyEvent.VK_SPACE) { //pause ja unpause
32              Pacman.paused = !Pacman.paused;
33          }
34      }
35      
36      public Player getPlayer() {
37          return player;
38      }
39  }
