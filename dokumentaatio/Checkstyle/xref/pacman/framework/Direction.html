3   /**
4    * Enum tarjoaa liikkumissuunnat pelin objekteille.
5    */
6   public enum Direction {
7       UP, LEFT, DOWN, RIGHT;
8       
9       /**
10       * Metodi kertoo onko parametrina annettava suunta vastakkainen tämän suunnan kanssa.
11       * 
12       * @param compare Verrattava suunta
13       * 
14       * @return totuusarvo onko verrattava vastakkainen
15       */
16      public boolean isOpposite(Direction compare) {
17          return compare == this.opposite();
18      }
19      
20      /**
21       * Metodi palauttaa vastakkaisen suunnan.
22       * 
23       * @return vastakkainen suunta
24       */
25      public Direction opposite() {
26          if (this == LEFT) {
27              return RIGHT;
28          } else if (this == DOWN) {
29              return UP;
30          } else if (this == RIGHT) {
31              return LEFT;
32          } else {
33              return DOWN;
34          }
35      }
36  }
